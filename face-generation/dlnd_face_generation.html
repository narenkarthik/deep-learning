Preprocess the Data¶

Since the project's main focus is on building the GANs, we'll preprocess the data for you. The values of the MNIST and CelebA dataset will be in the range of -0.5 to 0.5 of 28x28 dimensional images. The CelebA images will be cropped to remove parts of the image that don't include a face, then resized down to 28x28.

The MNIST images are black and white images with a single color channel while the CelebA images have 3 color channels (RGB color channel).

def discriminator(images, reuse=False):
    """
    Create the discriminator network
    :param image: Tensor of input image(s)
    :param reuse: Boolean if the weights should be reused
    :return: Tuple of (tensor output of the discriminator, tensor logits of the discriminator)
    """
    # TODO: Implement Function
    alpha = 0.2
    dropout = 0.2
    with tf.variable_scope('discriminator', reuse=reuse):
        layer1 = tf.layers.conv2d(images, 64, 5, strides=2, padding='same', kernel_initializer=tf.random_normal_initializer(stddev=0.05))
        layer1 = tf.maximum(alpha * layer1, layer1)

        layer2 = tf.layers.conv2d(layer1, 128, 5, strides=2, padding='same', kernel_initializer=tf.random_normal_initializer(stddev=0.05))
        layer2 = tf.layers.dropout(layer2, rate=dropout, training=True)        
        layer2 = tf.layers.batch_normalization(layer2, training=True)
        layer2 = tf.maximum(alpha * layer2, layer2)
        
        layer3 = tf.layers.conv2d(layer2, 256, 5, strides=2, padding='same', kernel_initializer=tf.random_normal_initializer(stddev=0.05))
        layer3 = tf.layers.dropout(layer3, rate=dropout, training=True)        
        layer3 = tf.layers.batch_normalization(layer3, training=True)
        layer3 = tf.maximum(alpha * layer3, layer3)
        
        flattened = tf.reshape(layer3, (-1, 4*4*256))
        logits = tf.layers.dense(flattened, 1, kernel_initializer=tf.random_normal_initializer(stddev=0.03))
        dis_output = tf.sigmoid(logits)
                
    return dis_output, logits

Generator¶

Implement generator to generate an image using z. This function should be able to reuse the variabes in the neural network. Use tf.variable_scope with a scope name of "generator" to allow the variables to be reused. The function should return the generated 28 x 28 x out_channel_dim images.

#observation based on same Hyper parameters in MNIST & CELEBA cells:
# 2-layer random normal initializer with strides 2 - generator loss were too high
# 3-layer random normal initializer with strides 1 - generator loss were not too high but didn't give a lot better image outputs
# 2-layer xavier initializer with strides 2 with output factored down - generator loss were within the expected range with better image outputs 
# 2-layer random normal initializer with strides 2 with output factored down - generator loss were within the expected range with best image outputs of all so used it for submission

def generator(z, out_channel_dim, is_train=True):
    """
    Create the generator network
    :param z: Input z
    :param out_channel_dim: The number of channels in the output image
    :param is_train: Boolean if generator is being used for training
    :return: The tensor output of the generator
    """
    # TODO: Implement Function
    #same alpha & dropout as discriminator
    alpha = 0.2
    dropout = 0.2
    with tf.variable_scope('generator', reuse=not is_train):
        layer1 = tf.layers.dense(z, 7 * 7 * 512)
        layer1 = tf.reshape(layer1, (-1, 7, 7, 512))
        layer1 = tf.layers.dropout(layer1, rate=dropout, training=True)        
        layer1 = tf.layers.batch_normalization(layer1, training=is_train)        
        layer1 = tf.maximum(alpha * layer1, layer1)
        #print("L1 shape: {}".format (layer1.shape))
        
        layer2 = tf.layers.conv2d_transpose(layer1, 256, 5, strides=2, padding='same', kernel_initializer=tf.random_normal_initializer(stddev=0.1))
        layer2 = tf.layers.dropout(layer2, rate=dropout, training=True)        
        layer2 = tf.layers.batch_normalization(layer2, training=is_train)
        layer2 = tf.maximum(alpha * layer2, layer2)
        #print("L2 shape: {}".format (layer2.shape))        
        
        logits = tf.layers.conv2d_transpose(layer2, out_channel_dim, 5, strides=2, padding='same', kernel_initializer=tf.random_normal_initializer(stddev=0.1))
        output = tf.tanh(logits)*0.7

        #print("output shape: {}".format (logits.shape))
        
    return output

def model_loss(input_real, input_z, out_channel_dim):
    """
    Get the loss for the discriminator and generator
    :param input_real: Images from the real dataset
    :param input_z: Z input
    :param out_channel_dim: The number of channels in the output image
    :return: A tuple of (discriminator loss, generator loss)
    """
    # TODO: Implement Function
    gmodel = generator(input_z, out_channel_dim)
    dmodel_real, dlogits_real = discriminator(input_real)
    dmodel_fake, dlogits_fake = discriminator(gmodel, reuse=True)

    dloss_real = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=dlogits_real, labels=tf.ones_like(dmodel_real) * 0.9))
    dloss_fake = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=dlogits_fake, labels=tf.zeros_like(dmodel_fake)))
    g_loss = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=dlogits_fake, labels=tf.ones_like(dmodel_fake)))

    d_loss = dloss_real + dloss_fake 
    
    return (d_loss, g_loss)

Optimization¶

def model_opt(d_loss, g_loss, learning_rate, beta1):
    """
    Get optimization operations
    :param d_loss: Discriminator loss Tensor
    :param g_loss: Generator loss Tensor
    :param learning_rate: Learning Rate Placeholder
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :return: A tuple of (discriminator training operation, generator training operation)
    """
    # TODO: Implement Function
    train_vars = tf.trainable_variables()
    d_vars = [var for var in train_vars if var.name.startswith('discriminator')]
    g_vars = [var for var in train_vars if var.name.startswith('generator')]

    opm = tf.get_collection(tf.GraphKeys.UPDATE_OPS)
    
    with tf.control_dependencies(opm):
        d_train_opm = tf.train.AdamOptimizer(learning_rate, beta1=beta1).minimize(d_loss, var_list=d_vars)
        g_train_opm = tf.train.AdamOptimizer(learning_rate, beta1=beta1).minimize(g_loss, var_list=g_vars)

    return d_train_opm, g_train_opm

import numpy as np

def show_generator_output(sess, n_images, input_z, out_channel_dim, image_mode):
    """
    Show example output for the generator
    :param sess: TensorFlow session
    :param n_images: Number of Images to display
    :param input_z: Input Z Tensor
    :param out_channel_dim: The number of channels in the output image
    :param image_mode: The mode to use for images ("RGB" or "L")
    """
    cmap = None if image_mode == 'RGB' else 'gray'
    z_dim = input_z.get_shape().as_list()[-1]
    example_z = np.random.uniform(-1, 1, size=[n_images, z_dim])

    samples = sess.run(
        generator(input_z, out_channel_dim, False),
        feed_dict={input_z: example_z})

    images_grid = helper.images_square_grid(samples, image_mode)
    pyplot.imshow(images_grid, cmap=cmap)
    pyplot.show()

def train(epoch_count, batch_size, z_dim, learning_rate, beta1, get_batches, data_shape, data_image_mode):
    """
    Train the GAN
    :param epoch_count: Number of epochs
    :param batch_size: Batch Size
    :param z_dim: Z dimension
    :param learning_rate: Learning Rate
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :param get_batches: Function to get batches
    :param data_shape: Shape of the data
    :param data_image_mode: The image mode to use for images ("RGB" or "L")
    """
    # TODO: Build Model
    input_real, input_z, input_lr = model_inputs(data_shape[1], data_shape[2], data_shape[3], z_dim)
    d_loss, g_loss = model_loss(input_real, input_z, data_shape[3])
    d_train_opm, g_train_opm = model_opt(d_loss, g_loss, input_lr, beta1) 
    
    steps = 0    
    
    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())
        for epoch_i in range(epoch_count):
            for batch_images in get_batches(batch_size):
                # TODO: Train Model
                steps += 1
                
                batch_z = np.random.uniform(-1, 1, size=(batch_size, z_dim))
                
                _ = sess.run(d_train_opm, feed_dict={input_real: batch_images, input_z: batch_z, input_lr: learning_rate})
                _ = sess.run(g_train_opm, feed_dict={input_real: batch_images, input_z: batch_z, input_lr: learning_rate})

                if steps % 100 == 0:      
                    dis_train_loss = d_loss.eval({input_z: batch_z, input_real: batch_images})
                    gen_train_loss = g_loss.eval({input_z: batch_z})

                    print("Epoch {}/{}...".format(epoch_i+1, epoch_count),
                          "Discriminator Loss: {:.4f}...".format(dis_train_loss),
                          "Generator Loss: {:.4f}".format(gen_train_loss))
                    
            show_generator_output(sess, 25, input_z, data_shape[3], data_image_mode)
